Week 3 Power BI Measures

Create these as New measures in Power BI:

Total Revenue = SUM(week3_dashboard_sales_model[Total_Sales])

Net Revenue = SUM(week3_dashboard_sales_model[Net_Sales])

Total Orders = DISTINCTCOUNT(week3_dashboard_sales_model[Order_ID])

Total Quantity Sold = SUM(week3_dashboard_sales_model[Quantity])

Average Order Value = DIVIDE([Total Revenue], [Total Orders])

Return Orders = CALCULATE(DISTINCTCOUNT(week3_dashboard_sales_model[Order_ID]), week3_dashboard_sales_model[Has_Return] = "Returned")

Return Rate = DIVIDE([Return Orders], [Total Orders])

Total Refund = SUM(week3_dashboard_sales_model[Total_Refund])

Stock Available = SUM(week3_inventory_model[Stock_Available])

Products Below Reorder Level = COUNTROWS(FILTER(week3_inventory_model, week3_inventory_model[Stock_Available] <= week3_inventory_model[Reorder_Level]))
